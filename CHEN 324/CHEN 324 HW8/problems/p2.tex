    \intertext{Fit $4^{\text{th}}$ order polynomials to equilibrium data in Appendix A.3-24. Then,}
    x_B &= f_1(x_A) \\
    x_C &= f_2(x_A) \\
    y_B &= f_3(y_A) \\
    y_C &= f_4(y_A) \\

    x_{A,0} L_0 &= x_{A,N} L_N + y_{A,1} V_1 \\
    x_{B,0} L_0 &= x_{B,N} L_N + y_{B,1} V_1 \\
    L_0 + V_{N+1} &= L_N + V_1 \\
    & x_{A,0} = 0.25 \\ 
    & x_{B,0} = 0.75 \\ 
    & x_{A,N} = 0.03 \\
    & y_{C,N+1} = 1 \\

\end{align*}
